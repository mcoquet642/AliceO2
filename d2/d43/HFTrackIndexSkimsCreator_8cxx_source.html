55  void process(aod::Collision const& collision,
56  soa::Join<aod::Tracks, aod::TracksCov, aod::TracksExtra> const& tracks)
57  {
58  math_utils::Point3D<float> vtxXYZ(collision.posX(), collision.posY(), collision.posZ());
59  for (auto& track : tracks) {
60  int status_2prong = 1; // selection flag
61  int status_3prong = 1; // selection flag
62 
63  if (b_dovalplots) {
64  hpt_nocuts->Fill(track.pt());
65  }
66 
67  // pT cut
68  if (track.pt() < ptmintrack_2prong) {
69  status_2prong = 0;
70  }
71  if (track.pt() < ptmintrack_3prong) {
72  status_3prong = 0;
73  }
74 
75  // eta cut
76  if (status_2prong && abs(track.eta()) > etamax_2prong) {
77  status_2prong = 0;
78  }
79  if (status_3prong && abs(track.eta()) > etamax_3prong) {
80  status_3prong = 0;
81  }
82 
83  // quality cut
84  if (doCutQuality && (status_2prong || status_3prong)) {
85  UChar_t clustermap = track.itsClusterMap();
86  bool isselected = track.tpcNClsFound() >= d_tpcnclsfound &&
87  track.flags() & o2::aod::track::ITSrefit &&
88  (TESTBIT(clustermap, 0) || TESTBIT(clustermap, 1));
89  if (!isselected) {
90  status_2prong = 0;
91  status_3prong = 0;
92  }
93  }
94 
95  // DCA cut
96  array<float, 2> dca;
97  if (status_2prong || status_3prong) {
98  auto trackparvar0 = getTrackParCov(track);
99  bool isprop = trackparvar0.propagateParamToDCA(vtxXYZ, d_bz, &dca, 100.); // get impact parameters
100  if (!isprop) {
101  status_2prong = 0;
102  status_3prong = 0;
103  }
104  if (status_2prong && abs(dca[0]) < dcatoprimxymin_2prong) {
105  status_2prong = 0;
106  }
107  if (status_3prong && abs(dca[0]) < dcatoprimxymin_3prong) {
108  status_3prong = 0;
109  }
110  }
111 
112  // fill histograms
113  if (b_dovalplots) {
114  if (status_2prong == 1) {
115  hpt_cuts_2prong->Fill(track.pt());
116  hdcatoprimxy_cuts_2prong->Fill(dca[0]);
117  heta_cuts_2prong->Fill(track.eta());
118  }
119  if (status_3prong == 1) {
120  hpt_cuts_3prong->Fill(track.pt());
121  hdcatoprimxy_cuts_3prong->Fill(dca[0]);
122  heta_cuts_3prong->Fill(track.eta());
123  }
124  }
125 
126  // fill table row
127  rowSelectedTrack(status_2prong, status_3prong, dca[0], dca[1]);
128  }
129  }
130 };
