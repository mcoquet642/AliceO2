177  template <typename T, typename U, typename V>
178  static double CPA(const T& posPV, const U& posSV, const array<V, 3>& mom)
179  {
180  // CPA = (l . p)/(|l| |p|)
181  auto lineDecay = array{posSV[0] - posPV[0], posSV[1] - posPV[1], posSV[2] - posPV[2]};
182  auto cos = dotProd(lineDecay, mom) / std::sqrt(mag2(lineDecay) * mag2(mom));
183  if (cos < -1.) {
184  return -1.;
185  }
186  if (cos > 1.) {
187  return 1.;
188  }
189  return cos;
190  }

197  template <std::size_t N, typename T, typename U, typename V>
198  static double CPAXY(const T& posPV, const U& posSV, const array<V, N>& mom)
199  {
200  // CPAXY = (r . pT)/(|r| |pT|)
201  auto lineDecay = array{posSV[0] - posPV[0], posSV[1] - posPV[1]};
202  auto momXY = array{mom[0], mom[1]};
203  auto cos = dotProd(lineDecay, momXY) / std::sqrt(mag2(lineDecay) * mag2(momXY));
204  if (cos < -1.) {
205  return -1.;
206  }
207  if (cos > 1.) {
208  return 1.;
209  }
210  return cos;
211  }

233  template <typename T, typename U, typename V>
234  static double CosThetaStar(const array<array<T, 3>, 2>& arrMom, const array<U, 2>& arrMass, V mTot, int iProng)
235  {
236  auto pVecTot = PVec(arrMom[0], arrMom[1]); // momentum of the mother particle
237  auto pTot = P(pVecTot); // magnitude of the momentum of the mother particle
238  auto eTot = E(pTot, mTot); // energy of the mother particle
239  auto gamma = eTot / mTot; // γ, Lorentz gamma factor of the mother particle
240  auto beta = pTot / eTot; // β, velocity of the mother particle
241  auto pStar = std::sqrt(sq(sq(mTot) - sq(arrMass[0]) - sq(arrMass[1])) - sq(2 * arrMass[0] * arrMass[1])) / (2 * mTot); // p*, prong momentum in the rest frame of the mother particle
242  // p* = √[(M^2 - m1^2 - m2^2)^2 - 4 m1^2 m2^2]/2M
243  // Lorentz transformation of the longitudinal momentum of the prong into the detector frame:
244  // p_L,i = γ (p*_L,i + β E*_i)
245  // p*_L,i = p_L,i/γ - β E*_i
246  // cos(θ*_i) = (p_L,i/γ - β E*_i)/p*
247  return (dotProd(arrMom[iProng], pVecTot) / (pTot * gamma) - beta * E(pStar, arrMass[iProng])) / pStar;
248  }

366  template <std::size_t N, typename T, typename U>
367  static double M2(const array<array<T, 3>, N>& arrMom, const array<U, N>& arrMass)
368  {
369  array<double, 3> momTotal{0., 0., 0.}; // candidate momentum vector
370  double energyTot{0.}; // candidate energy
371  for (auto iProng = 0; iProng < N; ++iProng) {
372  for (auto iMom = 0; iMom < 3; ++iMom) {
373  momTotal[iMom] += arrMom[iProng][iMom];
374  } // loop over momentum components
375  energyTot += E(arrMom[iProng], arrMass[iProng]);
376  } // loop over prongs
377  return energyTot * energyTot - P2(momTotal);
378  }

415  template <std::size_t N, typename T, typename U>
416  static bool isMCMatchedDecayRec(const T& particlesMC, const array<U, N>& arrDaughters, int PDGMother, array<int, N> arrPDGDaughters, bool acceptAntiParticles = false)
417  {
418  //Printf("MC Rec.: Expected mother PDG: %d", PDGMother);
419  int sgn = 1; // 1 if the expected mother is particle, -1 if antiparticle
420  int indexMother = -1; // index of the mother particle
421  int indexDaughterFirst = -1; // index of the first daughter
422  int indexDaughterLast = -1; // index of the last daughter
423  array<int, N> arrDaughtersIndex; // array of daughter indices
424  // loop over daughter particles
425  for (auto iProng = 0; iProng < N; ++iProng) {
426  auto particleI = arrDaughters[iProng].label(); // ith daughter particle
427  arrDaughtersIndex[iProng] = particleI.globalIndex();
428  auto indexParticleIMother = particleI.mother0();
429  if (indexParticleIMother == -1) {
430  //Printf("MC Rec. rejected: bad mother index: %d", indexParticleIMother);
431  return false;
432  }
433  // Get the mother.
434  auto particleIMother = particlesMC.iteratorAt(indexParticleIMother);
435  // Check mother's PDG code.
436  auto PDGParticleIMother = particleIMother.pdgCode(); // PDG code of the mother of the ith daughter
437  //Printf("MC Rec.: Daughter %d mother PDG: %d", iProng, PDGParticleIMother);
438  if (PDGParticleIMother != sgn * PDGMother) {
439  if (acceptAntiParticles && iProng == 0 && PDGParticleIMother == -PDGMother) {
440  sgn = -1; // PDG code of the first daughter's mother determines whether the expected mother is a particle or antiparticle.
441  } else {
442  //Printf("MC Rec. rejected: bad mother PDG: %s%d != %d", acceptAntiParticles ? "abs " : "", PDGParticleIMother, sgn * PDGMother);
443  return false;
444  }
445  }
446  // Check that all daughters have the same mother.
447  if (iProng == 0) {
448  indexMother = indexParticleIMother;
449  } else if (indexParticleIMother != indexMother) {
450  //Printf("MC Rec. rejected: different mothers: %d %d", indexMother, indexParticleIMother);
451  return false;
452  }
453  // Set the range of expected daughter indices.
454  if (iProng == 0) {
455  indexDaughterFirst = particleIMother.daughter0();
456  indexDaughterLast = particleIMother.daughter1();
457  // Check the daughter indices.
458  // Daughter indices are supposed to be consecutive.
459  if (indexDaughterLast <= indexDaughterFirst || indexDaughterFirst == -1 || indexDaughterLast == -1) {
460  //Printf("MC Rec. rejected: bad daughter index range: %d-%d", indexDaughterFirst, indexDaughterLast);
461  return false;
462  }
463  // Check the number of daughters.
464  if (indexDaughterLast - indexDaughterFirst + 1 != N) {
465  //Printf("MC Rec. rejected: incorrect number of daughters: %d", indexDaughterLast - indexDaughterFirst + 1);
466  return false;
467  }
468  }
469  // Check that the daughter is not a stepdaughter.
470  if (arrDaughtersIndex[iProng] < indexDaughterFirst || arrDaughtersIndex[iProng] > indexDaughterLast) {
471  //Printf("MC Rec. rejected: stepdaughter: %d outside %d-%d", arrDaughtersIndex[iProng], indexDaughterFirst, indexDaughterLast);
472  return false;
473  }
474  // Check that the daughter is not a twin.
475  for (auto jProng = 0; jProng < iProng; ++jProng) {
476  if (arrDaughtersIndex[iProng] == arrDaughtersIndex[jProng]) {
477  //Printf("MC Rec. rejected: twin daughter: %d", arrDaughtersIndex[iProng]);
478  return false;
479  }
480  }
481  // Check daughter's PDG code.
482  auto PDGParticleI = particleI.pdgCode(); // PDG code of the ith daughter
483  //Printf("MC Rec: Daughter %d PDG: %d", iProng, PDGParticleI);
484  bool PDGFound = false; // Is the PDG code of this daughter among the remaining expected PDG codes?
485  for (auto iProngCp = 0; iProngCp < N; ++iProngCp) {
486  if (PDGParticleI == sgn * arrPDGDaughters[iProngCp]) {
487  arrPDGDaughters[iProngCp] = 0; // Remove this PDG code from the array of expected ones.
488  PDGFound = true;
489  break;
490  }
491  }
492  if (!PDGFound) {
493  //Printf("MC Rec. rejected: bad daughter PDG: %d", PDGParticleI);
494  return false;
495  }
496  }
497  //Printf("MC Rec. accepted: m: %d, d: %d-%d", indexMother, indexDaughterFirst, indexDaughterLast);
498  return true;
499  }

520  template <std::size_t N, typename T, typename U>
521  static bool isMCMatchedDecayGen(const T& particlesMC, const U& candidate, int PDGParticle, array<int, N> arrPDGDaughters, bool acceptAntiParticles = false)
522  {
523  //Printf("MC Gen.: Expected particle PDG: %d", PDGParticle);
524  int sgn = 1; // 1 if the expected mother is particle, -1 if antiparticle
525  int indexDaughterFirst = 0; // index of the first daughter
526  int indexDaughterLast = 0; // index of the last daughter
527  // Check the PDG code of the particle.
528  auto PDGCandidate = candidate.pdgCode();
529  //Printf("MC Gen.: Candidate PDG: %d", PDGCandidate);
530  if (PDGCandidate != PDGParticle) {
531  if (acceptAntiParticles && PDGCandidate == -PDGParticle) {
532  sgn = -1;
533  } else {
534  //Printf("MC Gen. rejected: bad particle PDG: %s%d != %d", acceptAntiParticles ? "abs " : "", PDGCandidate, sgn * PDGParticle);
535  return false;
536  }
537  }
538  // Check the PDG codes of the decay products.
539  if (N > 1) {
540  //Printf("MC Gen.: Checking %d daughters", N);
541  // Set the range of expected daughter indices.
542  indexDaughterFirst = candidate.daughter0();
543  indexDaughterLast = candidate.daughter1();
544  // Check the daughter indices.
545  // Daughter indices are supposed to be consecutive.
546  if (indexDaughterLast <= indexDaughterFirst || indexDaughterFirst == -1 || indexDaughterLast == -1) {
547  //Printf("MC Gen. rejected: bad daughter index range: %d-%d", indexDaughterFirst, indexDaughterLast);
548  return false;
549  }
550  // Check the number of daughters.
551  if (indexDaughterLast - indexDaughterFirst + 1 != N) {
552  //Printf("MC Gen. rejected: incorrect number of daughters: %d", indexDaughterLast - indexDaughterFirst + 1);
553  return false;
554  }
555  // Check daughters' PDG codes.
556  for (auto indexDaughterI = indexDaughterFirst; indexDaughterI <= indexDaughterLast; ++indexDaughterI) {
557  auto candidateDaughterI = particlesMC.iteratorAt(indexDaughterI); // ith daughter particle
558  auto PDGCandidateDaughterI = candidateDaughterI.pdgCode(); // PDG code of the ith daughter
559  //Printf("MC Gen.: Daughter %d PDG: %d", indexDaughterI, PDGCandidateDaughterI);
560  bool PDGFound = false; // Is the PDG code of this daughter among the remaining expected PDG codes?
561  for (auto iProngCp = 0; iProngCp < N; ++iProngCp) {
562  if (PDGCandidateDaughterI == sgn * arrPDGDaughters[iProngCp]) {
563  arrPDGDaughters[iProngCp] = 0; // Remove this PDG code from the array of expected ones.
564  PDGFound = true;
565  break;
566  }
567  }
568  if (!PDGFound) {
569  //Printf("MC Gen. rejected: bad daughter PDG: %d", PDGCandidateDaughterI);
570  return false;
571  }
572  }
573  }
574  //Printf("MC Gen. accepted: m: %d, d: %d-%d", candidate.globalIndex(), indexDaughterFirst, indexDaughterLast);
575  return true;
576  }
