34 double ConvertPhiRange(double phi)
35 {
36  if (phi > M_PI) {
37  phi -= 2 * M_PI;
38  } else if (phi < -M_PI) {
39  phi += 2 * M_PI;
40  }
41 
42  return phi;
43 }

50 template <typename Track>
51 bool GetImpactParameterAndError(const Track& track, const o2df::VertexBase& primaryVertex, double& impactParameter,
52  double& impactParameterError)
53 {
54  impactParameter = -999.;
55  impactParameterError = -999.;
56 
57  o2df::DCA dca;
58  // FIXME: get this from CCDB
59  constexpr float magneticField{5.0}; // in kG
60  auto trackParameter = getTrackParCov(track);
61  bool propagate = trackParameter.propagateToDCA(primaryVertex, magneticField, &dca);
62 
63  if (propagate) {
64  impactParameter = 1000 * dca.getY();
65  impactParameterError = 1000 * std::sqrt(dca.getSigmaY2());
66  }
67  return propagate;
68 }

180  void process(const o2::soa::Join<o2::aod::Collisions, o2::aod::McCollisionLabels>::iterator& collision,
181  const o2::soa::Join<o2::aod::Tracks, o2::aod::TracksCov, o2::aod::McTrackLabels>& tracks,
182  const o2::aod::McParticles& mcParticles, const o2::aod::McCollisions& mcCollisions)
183  {
184  const o2df::VertexBase primaryVertex = getPrimaryVertex(collision);
185 
186  for (const auto& track : tracks) {
187  const double deltaPt = track.label().pt() - track.pt();
188  histos.fill("Pt/ptDiffMCRec", deltaPt);
189 
190  const double deltaPtOverPt = deltaPt / track.pt();
191  histos.fill("Pt/ptResolution", deltaPtOverPt);
192  histos.fill("Pt/ptResolutionVsPt", track.pt(), abs(deltaPtOverPt));
193  histos.fill("Pt/ptResolutionVsEta", track.eta(), abs(deltaPtOverPt));
194 
195  const double deltaEta = track.label().eta() - track.eta();
196  histos.fill("Eta/etaDiffMCReco", deltaEta);
197  histos.fill("Eta/etaDiffMCRecoVsEtaMC", deltaEta, track.label().eta());
198  histos.fill("Eta/etaDiffMCRecoVsEtaReco", deltaEta, track.eta());
199 
200  const auto deltaPhi = track_utils::ConvertPhiRange(track.label().phi() - track.phi());
201  histos.fill("Phi/phiDiffMCRec", deltaPhi);
202 
203  double impactParameter{-999.};
204  double impactParameterError{-999.};
205 
206  const bool propagate =
207  track_utils::GetImpactParameterAndError(track, primaryVertex, impactParameter, impactParameterError);
208 
209  if (propagate) {
210  histos.fill("ImpactParameter/impactParameterVsPt", track.pt(), impactParameter);
211  histos.fill("ImpactParameter/impactParameterVsEta", track.eta(), impactParameter);
212  histos.fill("ImpactParameter/impactParameterErrorVsPt", track.pt(), impactParameterError);
213  histos.fill("ImpactParameter/impactParameterErrorVsEta", track.eta(), impactParameterError);
214  }
215  }
216  }
217 };
